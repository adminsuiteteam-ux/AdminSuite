4. DATABASE SCHEMA (Firestore / PostgreSQL)

4.1 Firestore Collections (NoSQL)

```
users (collection)
  └─ {userId}
      ├─ email: string
      ├─ hashedPassword: string
      ├─ role: "admin" | "manager" | "hr"
      ├─ fullName: string
      ├─ avatarUrl: string (optional)
      ├─ createdAt: timestamp
      ├─ lastLogin: timestamp
      └─ isActive: boolean

employees (collection)
  └─ {employeeId}
      ├─ name: string
      ├─ address: string
      ├─ salary: number
      ├─ performance: number (1-5)
      ├─ status: "active" | "on_leave" | "terminated"
      ├─ department: string
      ├─ hireDate: timestamp
      ├─ customFields: map (dynamic key-value)
      ├─ createdBy: userId (admin)
      ├─ updatedBy: userId
      ├─ createdAt: timestamp
      └─ updatedAt: timestamp

clients (collection)
  └─ {clientId}
      ├─ clientName: string
      ├─ companyName: string
      ├─ email: string
      ├─ location: string
      ├─ website: string
      ├─ supportContact: string
      ├─ totalPaid: number (calculated aggregate)
      ├─ createdBy: userId
      ├─ createdAt: timestamp
      └─ updatedAt: timestamp

projects (collection)
  └─ {projectId}
      ├─ name: string
      ├─ specifications: string
      ├─ location: string
      ├─ startDate: timestamp
      ├─ endDate: timestamp (nullable)
      ├─ status: "planned" | "active" | "completed" | "on_hold"
      ├─ clientId: reference (clients)
      ├─ totalValue: number
      ├─ amountPaid: number (calculated)
      ├─ createdBy: userId (manager/admin)
      ├─ createdAt: timestamp
      └─ updatedAt: timestamp

transactions (collection)  # Finance
  └─ {transactionId}
      ├─ type: "income" | "expense"
      ├─ amount: number
      ├─ date: timestamp
      ├─ category: string
      ├─ description: string
      ├─ projectId: reference (nullable if non-project income)
      ├─ employeeId: reference (nullable if salary expense)
      ├─ paymentMethod: string (income only)
      ├─ receivedBy: string (income only)
      ├─ createdBy: userId (manager)
      ├─ approvedBy: userId (admin, optional)
      ├─ createdAt: timestamp
      └─ updatedAt: timestamp

customFields (collection)
  └─ {fieldId}
      ├─ module: "employee" | "project" | "finance"
      ├─ fieldName: string
      ├─ fieldType: "text" | "number" | "date" | "boolean" | "select"
      ├─ options: array (for select type)
      ├─ isRequired: boolean
      ├─ createdBy: userId (admin)
      ├─ createdAt: timestamp
      └─ isActive: boolean

notifications (collection)
  └─ {notificationId}
      ├─ userId: reference
      ├─ type: "info" | "warning" | "success" | "error"
      ├─ title: string
      ├─ message: string
      ├─ relatedId: string (employeeId, projectId, etc.)
      ├─ isRead: boolean
      ├─ createdAt: timestamp
      └─ expiresAt: timestamp (7 days TTL)

auditLogs (collection)
  └─ {logId}
      ├─ userId: reference
      ├─ action: "CREATE" | "UPDATE" | "DELETE" | "VIEW" | "EXPORT"
      ├─ module: "employee" | "client" | "project" | "finance" | "settings"
      ├─ recordId: string
      ├─ oldValues: map (optional)
      ├─ newValues: map (optional)
      ├─ ipAddress: string
      ├─ userAgent: string
      └─ timestamp: timestamp
```

4.2 PostgreSQL Schema (SQL Alternative)

```sql
-- Users table
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash TEXT NOT NULL,
    role VARCHAR(20) CHECK (role IN ('admin', 'manager', 'hr')),
    full_name VARCHAR(255),
    avatar_url TEXT,
    created_at TIMESTAMP DEFAULT NOW(),
    last_login TIMESTAMP,
    is_active BOOLEAN DEFAULT TRUE
);

-- Employees table
CREATE TABLE employees (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(255) NOT NULL,
    address TEXT,
    salary DECIMAL(10,2),
    performance INT CHECK (performance BETWEEN 1 AND 5),
    status VARCHAR(20) CHECK (status IN ('active', 'on_leave', 'terminated')),
    department VARCHAR(100),
    hire_date DATE,
    custom_fields JSONB,  -- For dynamic fields
    created_by UUID REFERENCES users(id),
    updated_by UUID REFERENCES users(id),
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- Clients table
CREATE TABLE clients (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    client_name VARCHAR(255) NOT NULL,
    company_name VARCHAR(255),
    email VARCHAR(255),
    location TEXT,
    website VARCHAR(255),
    support_contact VARCHAR(255),
    total_paid DECIMAL(10,2) DEFAULT 0,
    created_by UUID REFERENCES users(id),
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- Projects table
CREATE TABLE projects (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(255) NOT NULL,
    specifications TEXT,
    location TEXT,
    start_date DATE,
    end_date DATE,
    status VARCHAR(20) CHECK (status IN ('planned', 'active', 'completed', 'on_hold')),
    client_id UUID REFERENCES clients(id) ON DELETE SET NULL,
    total_value DECIMAL(10,2),
    amount_paid DECIMAL(10,2) DEFAULT 0,
    created_by UUID REFERENCES users(id),
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- Transactions table (Finance)
CREATE TABLE transactions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    type VARCHAR(10) CHECK (type IN ('income', 'expense')),
    amount DECIMAL(10,2) NOT NULL,
    date DATE NOT NULL,
    category VARCHAR(100),
    description TEXT,
    project_id UUID REFERENCES projects(id) ON DELETE SET NULL,
    employee_id UUID REFERENCES employees(id) ON DELETE SET NULL,
    payment_method VARCHAR(50),  -- for income
    received_by VARCHAR(255),    -- for income
    created_by UUID REFERENCES users(id),
    approved_by UUID REFERENCES users(id),
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- Custom Fields definition
CREATE TABLE custom_fields (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    module VARCHAR(20) CHECK (module IN ('employee', 'project', 'finance')),
    field_name VARCHAR(100) NOT NULL,
    field_type VARCHAR(20) CHECK (field_type IN ('text', 'number', 'date', 'boolean', 'select')),
    options JSONB,  -- for select type: ["option1", "option2"]
    is_required BOOLEAN DEFAULT FALSE,
    created_by UUID REFERENCES users(id),
    created_at TIMESTAMP DEFAULT NOW(),
    is_active BOOLEAN DEFAULT TRUE
);

-- Notifications table
CREATE TABLE notifications (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    type VARCHAR(20),
    title VARCHAR(255),
    message TEXT,
    related_id UUID,
    is_read BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT NOW(),
    expires_at TIMESTAMP DEFAULT NOW() + INTERVAL '7 days'
);

-- Audit Logs
CREATE TABLE audit_logs (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id),
    action VARCHAR(20),
    module VARCHAR(50),
    record_id UUID,
    old_values JSONB,
    new_values JSONB,
    ip_address INET,
    user_agent TEXT,
    timestamp TIMESTAMP DEFAULT NOW()
);

-- Indexes for performance
CREATE INDEX idx_users_role ON users(role);
CREATE INDEX idx_employees_status ON employees(status);
CREATE INDEX idx_projects_status ON projects(status);
CREATE INDEX idx_projects_client ON projects(client_id);
CREATE INDEX idx_transactions_date ON transactions(date);
CREATE INDEX idx_transactions_type ON transactions(type);
CREATE INDEX idx_notifications_user_read ON notifications(user_id, is_read);
```

```tsx
import React from 'react';
import { TouchableOpacity, Text, ActivityIndicator, View } from 'react-native';

interface ButtonProps {
  title: string;
  onPress: () => void;
  variant?: 'primary' | 'secondary' | 'danger' | 'outline';
  loading?: boolean;
  disabled?: boolean;
  fullWidth?: boolean;
  icon?: React.ReactNode;
}

export const Button: React.FC<ButtonProps> = ({
  title,
  onPress,
  variant = 'primary',
  loading = false,
  disabled = false,
  fullWidth = false,
  icon,
}) => {
  const getBgColor = () => {
    if (disabled) return 'bg-gray-300';
    switch (variant) {
      case 'primary': return 'bg-blue-600';
      case 'secondary': return 'bg-gray-600';
      case 'danger': return 'bg-red-600';
      case 'outline': return 'bg-transparent border-2 border-blue-600';
      default: return 'bg-blue-600';
    }
  };

  const getTextColor = () => {
    if (disabled) return 'text-gray-500';
    if (variant === 'outline') return 'text-blue-600';
    return 'text-white';
  };

  return (
    <TouchableOpacity
      onPress={onPress}
      disabled={disabled || loading}
      className={`${getBgColor()} ${fullWidth ? 'w-full' : 'px-4'} py-3 rounded-lg flex-row justify-center items-center`}
    >
      {icon && <View className="mr-2">{icon}</View>}
      {loading ? (
        <ActivityIndicator color={variant === 'outline' ? '#2563EB' : 'white'} />
      ) : (
        <Text className={`${getTextColor()} font-semibold text-center`}>{title}</Text>
      )}
    </TouchableOpacity>
  );
};
```

slices/authSlice.ts

```ts
import { createSlice, createAsyncThunk } from '@reduxjs/toolkit';
import AsyncStorage from '@react-native-async-storage/async-storage';
import { loginApi, verifyTokenApi } from '../../services/api/auth';

interface AuthState {
  user: null | {
    id: string;
    email: string;
    role: 'admin' | 'manager' | 'hr';
    fullName: string;
  };
  token: string | null;
  isLoading: boolean;
  error: string | null;
}

const initialState: AuthState = {
  user: null,
  token: null,
  isLoading: false,
  error: null,
};

export const login = createAsyncThunk(
  'auth/login',
  async ({ email, password }: { email: string; password: string }) => {
    const response = await loginApi(email, password);
    await AsyncStorage.setItem('token', response.token);
    return response;
  }
);

export const verifyToken = createAsyncThunk('auth/verifyToken', async () => {
  const token = await AsyncStorage.getItem('token');
  if (!token) throw new Error('No token');
  const user = await verifyTokenApi(token);
  return { user, token };
});

const authSlice = createSlice({
  name: 'auth',
  initialState,
  reducers: {
    logout: (state) => {
      state.user = null;
      state.token = null;
      AsyncStorage.removeItem('token');
    },
    clearError: (state) => {
      state.error = null;
    },
  },
  extraReducers: (builder) => {
    builder
      .addCase(login.pending, (state) => {
        state.isLoading = true;
        state.error = null;
      })
      .addCase(login.fulfilled, (state, action) => {
        state.isLoading = false;
        state.user = action.payload.user;
        state.token = action.payload.token;
      })
      .addCase(login.rejected, (state, action) => {
        state.isLoading = false;
        state.error = action.error.message || 'Login failed';
      })
      .addCase(verifyToken.fulfilled, (state, action) => {
        state.user = action.payload.user;
        state.token = action.payload.token;
      });
  },
});

export const { logout, clearError } = authSlice.actions;
export default authSlice.reducer;
```

13. CUSTOM FIELDS ENGINE (Dynamic Forms)

How it works

1. Admin creates a custom field definition in customFields collection/table.
2. Frontend fetches custom fields for each module on load.
3. Forms dynamically render additional inputs based on field definitions.
4. Values are stored as JSONB (PostgreSQL) or map (Firestore) in the main document.

Implementation

```tsx
// components/forms/DynamicField.tsx
import React from 'react';
import { View, Text, TextInput, Switch } from 'react-native';
import { Picker } from '@react-native-picker/picker';

interface CustomField {
  id: string;
  fieldName: string;
  fieldType: 'text' | 'number' | 'date' | 'boolean' | 'select';
  options?: string[];
  isRequired: boolean;
}

interface DynamicFieldProps {
  field: CustomField;
  value: any;
  onChange: (fieldId: string, value: any) => void;
  error?: string;
}

export const DynamicField: React.FC<DynamicFieldProps> = ({
  field,
  value,
  onChange,
  error,
}) => {
  const renderInput = () => {
    switch (field.fieldType) {
      case 'text':
        return (
          <TextInput
            value={value || ''}
            onChangeText={(text) => onChange(field.id, text)}
            className="border border-gray-300 rounded-lg p-3"
            placeholder={`Enter ${field.fieldName}`}
          />
        );
      case 'number':
        return (
          <TextInput
            value={value?.toString() || ''}
            onChangeText={(text) => onChange(field.id, parseFloat(text) || 0)}
            keyboardType="numeric"
            className="border border-gray-300 rounded-lg p-3"
          />
        );
      case 'boolean':
        return (
          <Switch
            value={value || false}
            onValueChange={(val) => onChange(field.id, val)}
          />
        );
      case 'select':
        return (
          <Picker
            selectedValue={value || ''}
            onValueChange={(itemValue) => onChange(field.id, itemValue)}
            className="border border-gray-300 rounded-lg"
          >
            <Picker.Item label="Select..." value="" />
            {field.options?.map((opt) => (
              <Picker.Item key={opt} label={opt} value={opt} />
            ))}
          </Picker>
        );
      case 'date':
        return (
          <TouchableOpacity
            onPress={() => {
              // Show date picker modal
            }}
            className="border border-gray-300 rounded-lg p-3"
          >
            <Text>{value ? new Date(value).toDateString() : 'Select date'}</Text>
          </TouchableOpacity>
        );
      default:
        return null;
    }
  };

  return (
    <View className="mb-4">
      <Text className="text-gray-700 font-medium mb-1">
        {field.fieldName} {field.isRequired && <Text className="text-red-500">*</Text>}
      </Text>
      {renderInput()}
      {error && <Text className="text-red-500 text-xs mt-1">{error}</Text>}
    </View>
  );
};
```

Appendix C: Finance Form (Add Expense)

```tsx
import React, { useState } from 'react';
import { View, Text, TextInput, TouchableOpacity, ScrollView } from 'react-native';
import { useDispatch } from 'react-redux';
import { createTransaction } from '../../store/slices/financeSlice';
import { Input } from '../../components/common/Input';
import { Button } from '../../components/common/Button';
import { Picker } from '@react-native-picker/picker';

export default function AddExpenseScreen({ navigation }) {
  const dispatch = useDispatch();
  const [form, setForm] = useState({
    amount: '',
    date: new Date().toISOString().split('T')[0],
    category: 'Software',
    description: '',
    projectId: '',
  });
  const [errors, setErrors] = useState({});

  const categories = ['Salary', 'Rent', 'Software', 'Office Supplies', 'Marketing', 'Travel', 'Other'];

  const validate = () => {
    const newErrors = {};
    if (!form.amount || parseFloat(form.amount) <= 0) newErrors.amount = 'Amount must be > 0';
    if (!form.description) newErrors.description = 'Description required';
    setErrors(newErrors);
    return Object.keys(newErrors).length === 0;
  };

  const handleSubmit = async () => {
    if (!validate()) return;
    await dispatch(createTransaction({ ...form, type: 'expense', amount: parseFloat(form.amount) }));
    navigation.goBack();
  };

  return (
    <ScrollView className="flex-1 bg-white p-4">
      <Text className="text-2xl font-bold mb-6">Add Expense</Text>

      <Input
        label="Amount"
        value={form.amount}
        onChangeText={(text) => setForm({ ...form, amount: text })}
        keyboardType="numeric"
        error={errors.amount}
        required
      />

      <View className="mb-4">
        <Text className="text-gray-700 font-medium mb-1">Date *</Text>
        <TouchableOpacity className="border border-gray-300 rounded-lg p-3">
          <Text>{form.date}</Text>
        </TouchableOpacity>
      </View>

      <View className="mb-4">
        <Text className="text-gray-700 font-medium mb-1">Category *</Text>
        <View className="border border-gray-300 rounded-lg">
          <Picker
            selectedValue={form.category}
            onValueChange={(itemValue) => setForm({ ...form, category: itemValue })}
          >
            {categories.map((cat) => (
              <Picker.Item key={cat} label={cat} value={cat} />
            ))}
          </Picker>
        </View>
      </View>

      <Input
        label="Description"
        value={form.description}
        onChangeText={(text) => setForm({ ...form, description: text })}
        multiline
        numberOfLines={3}
        error={errors.description}
        required
      />

      <Input
        label="Project (Optional)"
        value={form.projectId}
        onChangeText={(text) => setForm({ ...form, projectId: text })}
        placeholder="Link to project ID"
      />

      <Button title="Submit Expense" onPress={handleSubmit} variant="primary" fullWidth />
      
      <Button title="Cancel" onPress={() => navigation.goBack()} variant="outline" fullWidth />
    </ScrollView>
  );
}
```
